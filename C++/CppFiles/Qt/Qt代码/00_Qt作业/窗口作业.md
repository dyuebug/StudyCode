# 第二题
```
创建一个程序，存在一个窗口W，W的大小是800*600，存在两个QPushButton子类对象A和B，位置是0，0 和 400，150。使用信号槽机制和事件机制，当在鼠标在A上按住并移动时，获取鼠标位置x和y，并将B的位置改成10*x+400, 10*y+150
```
mainwindow.h
```C++
#ifndef MAINWINDOW_H
#define MAINWINDOW_H

#include <QMainWindow>
#include <QPushButton>
#include <QMouseEvent>
QT_BEGIN_NAMESPACE
namespace Ui { class MainWindow; }
QT_END_NAMESPACE

class MainWindow : public QMainWindow
{
    Q_OBJECT

public:
    MainWindow(QWidget *parent = nullptr);
    ~MainWindow();
    void mouseMoveEvent(QMouseEvent *event) override;
    bool eventFilter(QObject *obj, QEvent *event) override;

private:
    Ui::MainWindow *ui;
    QPushButton *buttonA;
    QPushButton *buttonB;
    bool dragging;
};
#endif // MAINWINDOW_H

```

mainwindow.cpp
```C++
#include "mainwindow.h"
#include "ui_mainwindow.h"
#include <QMouseEvent>
#include <QDebug>
MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent)
    ,dragging(false)
{
    // 设置窗口大小
    setFixedSize(800, 600);
    // 创建按钮  并设置位置
    buttonA = new QPushButton("A", this);
    buttonA->setGeometry(0, 0, 100, 50);

    buttonB = new QPushButton("B", this);
    buttonB->setGeometry(400, 150, 100, 50);

    // 安装事件过滤器到按钮 A
    buttonA->installEventFilter(this);
}

MainWindow::~MainWindow() {}

void MainWindow::mouseMoveEvent(QMouseEvent *event) {
    if (dragging) {
         // 如果正在拖动，则更新按钮 B 的位置
        int x = event->pos().x();
        int y = event->pos().y();
        qDebug() << "Mouse Move Event - Position:" << x << y; // 打印鼠标位置
        buttonB->move(10 * x + 400, 10 * y + 150);
    }
    QMainWindow::mouseMoveEvent(event);
}

bool MainWindow::eventFilter(QObject *obj, QEvent *event) {
    if (obj == buttonA) {
         // 处理按钮 A 的事件
        if (event->type() == QEvent::MouseButtonPress) {
            QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
            if (mouseEvent->button() == Qt::LeftButton) {
                dragging = true;// 设置拖动标志
                qDebug() << "Mouse Button Press - Position:" << mouseEvent->pos().x() << mouseEvent->pos().y(); // 打印鼠标位置
                return true;// 事件已处理
            }
        } else if (event->type() == QEvent::MouseButtonRelease) {
            // 处理鼠标释放事件
            dragging = false; // 取消拖动标志
            QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
            qDebug() << "Mouse Button Release - Position:" << mouseEvent->pos().x() << mouseEvent->pos().y(); // 打印鼠标位置
            return true;// 事件已处理
        } else if (event->type() == QEvent::MouseMove) {
            // 处理鼠标移动事件
            QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
            if (dragging) {
                int x = mouseEvent->pos().x();
                int y = mouseEvent->pos().y();
                qDebug() << "Mouse Move Event (Dragging) - Position:" << x << y; // 打印鼠标位置
                buttonB->move(10 * x + 400, 10 * y + 150);
                return true;
            }
        }
    }
    return QMainWindow::eventFilter(obj, event);
}

# 第3题
```
创建一个界面W，上面有QPushButton对象A,B,C,D,E,F：

- 将A作为根，BCD是A的孩子，E、F是B的孩子：
- 在E上触发鼠标按下事件，传递给W处理
- 在所有对象上触发鼠标移动事件，显示当前鼠标位置
- 进入E和F对象之后，C会移动到另一个位置
- 只要C移动，D也会移动
```

#include "mainwindow.h"
#include "ui_mainwindow.h"
#include <QDebug>
MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent)

{
    setFixedSize(800, 600);

    // 创建根按钮 A
    buttonA = new QPushButton("A", this);
    buttonA->setGeometry(100, 100, 100, 50);

    // 创建子按钮 B, C, D，设置为 A 的子按钮
    buttonB = new QPushButton("B", this);
    buttonC = new QPushButton("C", this);
    buttonD = new QPushButton("D", this);

    // 创建子按钮 E, F，设置为 B 的子按钮
    buttonE = new QPushButton("E", this);
    buttonF = new QPushButton("F", this);

    // 设置按钮的位置
    buttonB->setGeometry(210, 100, 100, 50);
    buttonC->setGeometry(320, 100, 100, 50);
    buttonD->setGeometry(430, 100, 100, 50);
    buttonE->setGeometry(540, 100, 100, 50);
    buttonF->setGeometry(650, 100, 100, 50);

    // 安装事件过滤器
    buttonA->installEventFilter(this);
    buttonB->installEventFilter(this);
    buttonC->installEventFilter(this);
    buttonD->installEventFilter(this);
    buttonE->installEventFilter(this);
    buttonF->installEventFilter(this);

}

MainWindow::~MainWindow()
{

}

bool MainWindow::eventFilter(QObject *obj, QEvent *event) {
    // 处理鼠标移动事件，显示当前鼠标位置
    if (event->type() == QEvent::MouseMove) {
        QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
        int x = mouseEvent->pos().x();
        int y = mouseEvent->pos().y();
        qDebug() << "Mouse Move Event - Position:" << x << y;
    }

    // 处理鼠标按下事件，特别是在按钮 E 上
    if (obj == buttonE && event->type() == QEvent::MouseButtonPress) {
        QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
        if (mouseEvent->button() == Qt::LeftButton) {
            qDebug() << "Mouse Button Pressed on E";
            // 传递事件给主窗口
            QMouseEvent *newEvent = new QMouseEvent(*mouseEvent);
            QCoreApplication::postEvent(this, newEvent);
            return true; // 事件已处理
        }
    }

    // 处理进入 E 和 F 对象时，移动 C
    if ((obj == buttonE || obj == buttonF) && event->type() == QEvent::Enter) {
        buttonC->move(300, 300); // 移动 C 到另一个位置
    }

    // 处理 C 移动事件，移动 D
    if (obj == buttonC && event->type() == QEvent::Move) {
        buttonD->move(buttonC->x() + 100, buttonC->y()); // 将 D 移动到相对于 C 的新位置
    }

    return QMainWindow::eventFilter(obj, event);
}

# 第4题
```
实现一个“打蚊子”游戏

在屏幕中央有一个600*400的QWidget，一个用来统计分数的QLabel

一开始会在QWidget内部随机位置生成一个蚊子，当鼠标点击到蚊子以后，旧蚊子消失然后在另一个位置生成新的蚊子，分数增加。

提示：这里需要继承QWidget类然后重写paintEvent，在paintEvent当中可以创建QPainter对象绘制各种图形和图片
```
mainwindow.h
```C++

    ~MainWindow();

public slots:
    void increaseScore();

private:
    QLabel *scoreLabel;
    int score;
    MosquitoWidget *gameWidget;
};

#endif // MAINWINDOW_H

```

mainwindow.cpp
```C++
#include "mainwindow.h"
#include "mosquitowidget.h"
MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent), score(0) {
    // 设置主窗口大小
    setFixedSize(800, 600);

    // 创建分数显示标签
    scoreLabel = new QLabel("Score: 0", this);
    scoreLabel->setGeometry(10, 10, 100, 30);

    // 创建游戏部件
    gameWidget = new MosquitoWidget(this);
    gameWidget->setGeometry(100, 100, 600, 400);

    // 连接信号和槽
    connect(gameWidget, &MosquitoWidget::mosquitoClicked, this, &MainWindow::increaseScore);
}

MainWindow::~MainWindow() {}

void MainWindow::increaseScore() {
    score++;
    scoreLabel->setText(QString("Score: %1").arg(score));
}

mosquitowidget.h
```C++
#ifndef MOSQUITOWIDGET_H
#define MOSQUITOWIDGET_H

#include <QWidget>

class MosquitoWidget : public QWidget {
    Q_OBJECT

public:
    MosquitoWidget(QWidget *parent = nullptr);
    ~MosquitoWidget();

signals:
    void mosquitoClicked();

protected:
    void paintEvent(QPaintEvent *event) override;
    void mousePressEvent(QMouseEvent *event) override;

private slots:
    void generateNewMosquito();

};

#endif // MOSQUITOWIDGET_H

```
mosquitowidget.cpp
```C++
#include "mosquitowidget.h"
#include <QPainter>
#include <QMouseEvent>
#include <QRandomGenerator>

MosquitoWidget::MosquitoWidget(QWidget *parent)
    : QWidget(parent), timer(new QTimer(this)) {
    setFixedSize(600, 400);

    connect(timer, &QTimer::timeout, this, &MosquitoWidget::generateNewMosquito);
    timer->start(1000); // 每秒生成一个新的蚊子

    generateNewMosquito();
}

MosquitoWidget::~MosquitoWidget() {}

void MosquitoWidget::paintEvent(QPaintEvent *event) {
    QPainter painter(this);
    painter.setBrush(Qt::black);
    painter.drawEllipse(mosquitoPosition, 10, 10); // 绘制蚊子（一个黑色圆点）
}

void MosquitoWidget::mousePressEvent(QMouseEvent *event) {
    if ((event->pos() - mosquitoPosition).manhattanLength() <= 10) {
        emit mosquitoClicked();
        generateNewMosquito();
    }
}

void MosquitoWidget::generateNewMosquito() {
    int x = QRandomGenerator::global()->bounded(width());
    int y = QRandomGenerator::global()->bounded(height());
    mosquitoPosition = QPoint(x, y);
    update(); // 触发重新绘制
}

#include "mainwindow.h"

MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent) {
    QWidget *centralWidget = new QWidget(this);
    setCentralWidget(centralWidget);

    // 创建主布局并添加到中央窗口
    QVBoxLayout *mainLayout = new QVBoxLayout(centralWidget);
    mainLayout->addWidget(createTypeGroup()); // 添加窗口类型选项组
    mainLayout->addWidget(createHintsGroup()); // 添加窗口提示选项组

    // 创建并添加预览按钮
    previewButton = new QPushButton("Preview", this);
    mainLayout->addWidget(previewButton);

    // 连接预览按钮的点击信号到显示预览窗口的槽
    connect(previewButton, &QPushButton::clicked, this, &MainWindow::showPreview);
}

MainWindow::~MainWindow() {}

QGroupBox *MainWindow::createTypeGroup() {
    QGroupBox *groupBox = new QGroupBox("Type");

    // 创建窗口类型选项
    windowRadioButton = new QRadioButton("Window");
    dialogRadioButton = new QRadioButton("Dialog");
    sheetRadioButton = new QRadioButton("Sheet");
    drawerRadioButton = new QRadioButton("Drawer");
    popupRadioButton = new QRadioButton("Popup");
    toolRadioButton = new QRadioButton("Tool");
    tooltipRadioButton = new QRadioButton("Tooltip");
    splashScreenRadioButton = new QRadioButton("Splash screen");

    // 默认选中 Window 类型
    windowRadioButton->setChecked(true);

    // 将窗口类型选项添加到布局中
    QVBoxLayout *vbox = new QVBoxLayout;
    vbox->addWidget(windowRadioButton);
    vbox->addWidget(dialogRadioButton);
    vbox->addWidget(sheetRadioButton);
    vbox->addWidget(drawerRadioButton);
    vbox->addWidget(popupRadioButton);
    vbox->addWidget(toolRadioButton);
    vbox->addWidget(tooltipRadioButton);
    vbox->addWidget(splashScreenRadioButton);
    groupBox->setLayout(vbox);

    return groupBox;
}

QGroupBox *MainWindow::createHintsGroup() {
    QGroupBox *groupBox = new QGroupBox("Hints");

    // 创建窗口提示选项
    msWindowsFixedSizeDialogCheckBox = new QCheckBox("MS Windows fixed size dialog");
    x11BypassWindowManagerCheckBox = new QCheckBox("X11 bypass window manager");
    framelessWindowCheckBox = new QCheckBox("Frameless window");
    noDropShadowCheckBox = new QCheckBox("No drop shadow");
    windowTitleCheckBox = new QCheckBox("Window title");
    windowTitleCheckBox->setChecked(true);
    windowSystemMenuCheckBox = new QCheckBox("Window system menu");
    windowSystemMenuCheckBox->setChecked(true);
    customizeWindowCheckBox = new QCheckBox("Customize window");
    customizeWindowCheckBox->setChecked(true);

    // 将窗口提示选项添加到网格布局中
    QGridLayout *grid = new QGridLayout;
    grid->addWidget(msWindowsFixedSizeDialogCheckBox, 0, 0);
    grid->addWidget(x11BypassWindowManagerCheckBox, 0, 1);
    grid->addWidget(framelessWindowCheckBox, 1, 0);
    grid->addWidget(noDropShadowCheckBox, 1, 1);
    grid->addWidget(windowTitleCheckBox, 2, 0);
    grid->addWidget(windowSystemMenuCheckBox, 2, 1);
    grid->addWidget(customizeWindowCheckBox, 3, 0);
    groupBox->setLayout(grid);

    return groupBox;

void MainWindow::showPreview() {
    QDialog previewDialog(this);
    previewDialog.setWindowTitle("Preview");

    // 设置窗口类型
    Qt::WindowFlags flags;
    if (windowRadioButton->isChecked())
        flags = Qt::Window;
    else if (dialogRadioButton->isChecked())
        flags = Qt::Dialog;
    else if (sheetRadioButton->isChecked())
        flags = Qt::Sheet;
    else if (drawerRadioButton->isChecked())
        flags = Qt::Drawer;
    else if (popupRadioButton->isChecked())
        flags = Qt::Popup;
    else if (toolRadioButton->isChecked())
        flags = Qt::Tool;
    else if (tooltipRadioButton->isChecked())
        flags = Qt::ToolTip;
    else if (splashScreenRadioButton->isChecked())
        flags = Qt::SplashScreen;

    // 设置窗口提示
    if (msWindowsFixedSizeDialogCheckBox->isChecked())
        flags |= Qt::MSWindowsFixedSizeDialogHint;
    if (x11BypassWindowManagerCheckBox->isChecked())
        flags |= Qt::X11BypassWindowManagerHint;
    if (framelessWindowCheckBox->isChecked())
        flags |= Qt::FramelessWindowHint;
    if (noDropShadowCheckBox->isChecked())
        flags |= Qt::NoDropShadowWindowHint;
    if (windowTitleCheckBox->isChecked())
        flags |= Qt::WindowTitleHint;
    if (windowSystemMenuCheckBox->isChecked())
        flags |= Qt::WindowSystemMenuHint;
    if (customizeWindowCheckBox->isChecked())
        flags |= Qt::CustomizeWindowHint;

    previewDialog.setWindowFlags(flags);

    // 创建并设置预览窗口布局
    QVBoxLayout *layout = new QVBoxLayout(&previewDialog);
    QLabel *label = new QLabel("Qt::Sheet\nQt::WindowTitleHint\nQt::WindowSystemMenuHint\nQt::CustomizeWindowHint", &previewDialog);
    layout->addWidget(label);
    QPushButton *closeButton = new QPushButton("Close", &previewDialog);
    layout->addWidget(closeButton);

    // 连接关闭按钮的点击信号到预览窗口的 accept 槽
    connect(closeButton, &QPushButton::clicked, &previewDialog, &QDialog::accept);

    previewDialog.exec();
}
