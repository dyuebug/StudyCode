## mainwindow.h
```C++
#ifndef MAINWINDOW_H
#define MAINWINDOW_H

#include <QMainWindow>
#include <QDebug>
#include <QEvent>
#include <QMouseEvent>
QT_BEGIN_NAMESPACE
namespace Ui { class MainWindow; }
QT_END_NAMESPACE

class MainWindow : public QMainWindow
{
    Q_OBJECT

public:
    MainWindow(QWidget *parent = nullptr);
    ~MainWindow();
    bool event(QEvent *ev) override{
        if(ev->type() == QEvent::MouseButtonPress){
            qDebug() << "MainWindow event press";
        }
        return QMainWindow::event(ev);
    }
    void mousePressEvent(QMouseEvent *ev) override{
        qDebug() << "MainWindow event_handler press";
        ev->ignore();
    }
private:
    Ui::MainWindow *ui;
};
#endif // MAINWINDOW_H
```

## main.cpp 
```C++
#include "mainwindow.h"
#include "out.h"
#include <QApplication>

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);
    MainWindow w;
    w.resize(800,600);
    Out *pout = new Out(&w);
    pout->resize(700,500);
    pout->move(50,50);
    pout->setFrameStyle(QFrame::Panel | QFrame::Plain);
    pout->setLineWidth(2);
    Mid *pmid = new Mid(pout);
    pmid->resize(600,400);
    pmid->move(50,50);
    pmid->setFrameStyle(QFrame::Panel | QFrame::Plain);
    pmid->setLineWidth(2);
    In *pin = new In(pmid);
    pin->resize(500,300);
    pin->move(50,50);
    pin->setFrameStyle(QFrame::Panel | QFrame::Plain);
    pin->setLineWidth(2);
    w.show();
    return a.exec();
}
```
