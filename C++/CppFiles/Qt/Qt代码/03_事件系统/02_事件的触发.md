## mylabel.h
```C++
#ifndef MYLABEL_H
#define MYLABEL_H

#include <QObject>
#include <QtWidgets/QtWidgets>
class MyLabel : public QLabel
{
    Q_OBJECT
public:
    explicit MyLabel(QWidget *parent = nullptr);
    void mouseReleaseEvent(QMouseEvent *ev) override;
signals:
    void myClicked();
};

#endif // MYLABEL_H
```

## mylabel.cpp
```C++
#include "mylabel.h"

MyLabel::MyLabel(QWidget *parent)
    : QLabel(parent)
{

}
void MyLabel::mouseReleaseEvent(QMouseEvent *ev){
    qDebug() << "mouse release";
    emit myClicked();
}

```

## mainwindow.h
```C++
#ifndef MAINWINDOW_H
#define MAINWINDOW_H

#include <QMainWindow>

QT_BEGIN_NAMESPACE
namespace Ui { class MainWindow; }
QT_END_NAMESPACE

class MainWindow : public QMainWindow
{
    Q_OBJECT

public:
    MainWindow(QWidget *parent = nullptr);
    ~MainWindow();
    int times;
private:
    Ui::MainWindow *ui;
public slots:
    void do_change_times();
};
#endif // MAINWINDOW_H
```

## mainwindow.cpp
```C++
#include "mainwindow.h"
#include "ui_mainwindow.h"

MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent)
    , times(0)
    , ui(new Ui::MainWindow)
{
    ui->setupUi(this);
}

MainWindow::~MainWindow()
{
    delete ui;
}
void MainWindow::do_change_times(){
    ++times;
    ui->timesLabel->setText(QString("clicked times：%1").arg(times));
}
```

## main.cpp
```C++
#include "mainwindow.h"
#include "mylabel.h"
#include <QApplication>

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);
    MainWindow w;
    w.resize(1024,768);
    MyLabel * label = new MyLabel(&w);
    label->resize(300,400);
    QPixmap pixmap(":/new/prefix1/test1.png");
    label->setPixmap(pixmap);
    QObject::connect(label,&MyLabel::myClicked,&w,&MainWindow::do_change_times);
    w.show();
    return a.exec();
}
```
