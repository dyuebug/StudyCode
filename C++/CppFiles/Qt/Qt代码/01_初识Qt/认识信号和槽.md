## mainwindow.h
```c++
#ifndef MAINWINDOW_H
#define MAINWINDOW_H

#include <QMainWindow>
#include <QPushButton>
QT_BEGIN_NAMESPACE
namespace Ui { class MainWindow; }
QT_END_NAMESPACE

class MainWindow : public QMainWindow
{
    Q_OBJECT

public:
    MainWindow(QWidget *parent = nullptr);
    ~MainWindow();

private slots:
    void on_pushButton_clicked();
private slots:
    void do_alterBtn_clicked(); // do_ 说明是自定义的槽函数
private:
    Ui::MainWindow *ui;
    QPushButton * newBtn;
};
#endif // MAINWINDOW_H

```

## ## mainwindow.cpp
```c++
#include "mainwindow.h"
#include "ui_mainwindow.h"

MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent)
    , ui(new Ui::MainWindow)
{
    ui->setupUi(this);
    //自定义槽要进行连接信号
    QObject::connect(ui->alterBtn,SIGNAL(clicked()),this,SLOT(do_alterBtn_clicked()));
}

MainWindow::~MainWindow()
{
    delete ui;
}
void MainWindow::on_pushButton_clicked(){
    QString text = ui->label->text();
    if(text == "hello"){
        ui->label->setText("world");
    }
    else{
        ui->label->setText("hello");
    }
}
void MainWindow::do_alterBtn_clicked(){
    QString text = ui->label->text();
    if(text == "hello"){
        ui->label->setText("HELLO");
    }
    else if(text == "world"){
        ui->label->setText("WORLD");
    }
}


```
