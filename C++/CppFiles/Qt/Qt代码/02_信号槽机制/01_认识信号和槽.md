## b.h
```c++
#ifndef B_H
#define B_H

#include <QObject>

class B : public QObject
{
    Q_OBJECT
public:
    explicit B(QObject *parent = nullptr);

public slots:
    void mySlot();
    void mySlot1();
};

#endif // B_H

```
## b.cpp
```c++
#include "b.h"
#include <QDebug>
#include <QtWidgets/QtWidgets> // 万能头文件
B::B(QObject *parent) : QObject(parent)
{

}
void B::mySlot(){
    qDebug() << "before I am slot";
    Sleep(1000);
    qDebug() << "after I am slot";
}
void B::mySlot1(){
    qDebug() << "before I am slot1";
    Sleep(1000);
    qDebug() << "after I am slot1";
}

```
## main.cpp
```c++
#include "mainwindow.h"
#include "a.h"
#include "b.h"
#include <QApplication>
#include <QDebug>
int main(int argc, char *argv[])
{
    QApplication app(argc, argv);
    A a;
    B b;
//    QObject::connect(&a,SIGNAL(mySignal()),
//                     &b,SLOT(mySlot()));
    QObject::connect(&a,&A::mySignal,
                     &b,&B::mySlot1);
    QObject::connect(&a,&A::mySignal,
                     &b,&B::mySlot);
    QObject::connect(&a,&A::mySignal,
                     &b,&B::mySlot1);
    emit a.mySignal();
    qDebug() << "after mySignal";
    return app.exec();
}

```
