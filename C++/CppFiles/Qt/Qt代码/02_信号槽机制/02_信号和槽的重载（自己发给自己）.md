## a.h
```c++
#ifndef A_H
#define A_H

#include <QObject>

class A : public QObject
{
    Q_OBJECT
public:
    explicit A(QObject *parent = nullptr);

signals:
    void mySignal(int value);
    void mySignal();
private slots:
    void mySlot(int value1,int value2);
    void mySlot(int value);
    void mySlot();
};

#endif // A_H

```
## a.cpp
```c++
#include "a.h"
#include <QDebug>
A::A(QObject *parent) : QObject(parent){
//    QObject::connect(this,QOverload<>::of(&A::mySignal),
//                     this,QOverload<>::of(&A::mySlot));
    QObject::connect(this,QOverload<int>::of(&A::mySignal),
                     this,QOverload<int>::of(&A::mySlot));
    QObject::connect(this,QOverload<int>::of(&A::mySignal),
                     this,QOverload<int>::of(&A::mySlot));
}
void A::mySlot(int value){
    qDebug() << "I am slot one value = " << value;
}
void A::mySlot(){
    qDebug() << "I am slot zero";
}
void A::mySlot(int value1,int value2){
    qDebug() << "I am slot two value1 = " << value1 << " value2 = " << value2;
}

```
## main.cpp
```c++
#include "mainwindow.h"
#include "a.h"
#include <QApplication>

int main(int argc, char *argv[])
{
    QApplication app(argc, argv);
    A a;
    emit a.mySignal(123);
//    emit a.mySignal();
    return app.exec();
}
```
